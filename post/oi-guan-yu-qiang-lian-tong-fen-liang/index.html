#include <bits/stdc++.h>
using namespace std;
const int N = 2e5+10;
int n,m,fa[N],cnt,ans,u1,v1;
struct Edge
{
        int u,v,w;
};
Edge e[N];
bool cmp(Edge x, Edge y)
{
        return x.w<y.w;
}
int find(int x)
{
        while (x!=fa[x]) x=fa[x]=fa[fa[x]];
        return x;
}
void kruskal()
{
        sort(e+1,e+m+1,cmp);
        for (int i=1; i<=m; i++)
        {
                u1=find(e[i].u),v1=find(e[i].v);
                if (u1==v1) continue;
                ans+=e[i].w;
                fa[v1]=u1;
                if (++cnt==n-1) break;
        }
}
int main()
{
        cin >> n >> m;
        for (int i=1; i<=n; i++) fa[i]=i;
        for (int i=1; i<=m; i++)
                cin >> e[i].u >> e[i].v >> e[i].w;
        kruskal();
        cout << ans;
        return 0;
}
